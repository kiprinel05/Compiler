int factorial(int n) {
    int result = 1;
    while (n > 1) {
        result = result * n;
        n = n - 1;
    }
    return result;
}

float calculateAverage(int a, int b, int c) {
    int sum = a + b + c;
    return sum / 3.0;
}

    int x = 5;
